41pub fn run(
42	path: Option<std::path::PathBuf>,
43	name: Option<String>,
44	url: Option<String>,
45	mut languages: Vec<String>,
46	content_rating: Option<SourceContentRating>,
47	template: bool,
48	template_name: Option<String>,
49) -> anyhow::Result<()> {
50	// verify already input languages
51	for lang in &languages {
52		if !VALID_LANGUAGE_CODES.contains(&lang.as_str()) {
53			return Err(anyhow!("Invalid language code: {}", lang));
54		}
55	}
56
57	let current_dir = std::env::current_dir().context("Failed to get current directory")?;
58	let target_path = path.unwrap_or(current_dir.clone());
59
60	// use directory name as default source name
61	let directory_name = target_path
62		.file_name()
63		.context("Failed to get directory name")?
64		.to_string_lossy();
65
66	let template_name = if template {
67		Some(match template_name {
68			Some(n) => n,
69			None => Input::<String>::new()
70				.with_prompt("Template name")
71				.interact_text()?,
72		})
73	} else {
74		None
75	};
76
77	// if name is None, prompt for it. use directory name as default
78	let name = match name {
79		Some(n) => n,
80		None => Input::new()
81			.with_prompt("Source name")
82			.default(directory_name.to_string())
83			.interact_text()?,
84	};
85
86	// if url is None, prompt for it
87	let url = match url {
88		Some(u) => u,
89		None => Input::new()
90			.with_prompt("Source URL")
91			.validate_with(|input: &String| -> Result<(), &str> {
92				if input.starts_with("http://") || input.starts_with("https://") {
93					Ok(())
94				} else {
95					Err("URL must start with http:// or https://")
96				}
97			})
98			.interact_text()?,
99	};
100
101	// get language codes
102	if languages.is_empty() {
103		let available_languages = [
104			("English", "en"),
105			("Chinese", "zh"),
106			("Spanish", "es"),
107			("Russian", "ru"),
108			("Vietnamese", "vi"),
109			("Other", ""),
110		];
111
112		let language_names: Vec<&str> = available_languages.iter().map(|(name, _)| *name).collect();
113
114		let selections = MultiSelect::new()
115			.with_prompt("Select supported languages")
116			.items(&language_names)
117			.interact()?;
118
119		if selections.is_empty() {
120			return Err(anyhow!("You must select at least one language"));
121		}
122
123		for i in selections {
124			if language_names[i] == "Other" {
125				let other_input: String = Input::new()
126					.with_prompt("Enter additional language codes (e.g. `id pt ja`)")
127					.validate_with(|input: &String| -> Result<(), &str> {
128						let codes = input.split_whitespace().map(|s| s.to_lowercase());
129						for code in codes {
130							if !VALID_LANGUAGE_CODES.contains(&code.as_str()) {
131								return Err(
132									"One or more language codes are not valid ISO 639-2 codes",
133								);
134							}
135						}
136						Ok(())
137					})
138					.interact_text()?;
139
140				let mut other_codes: Vec<String> = other_input
141					.split_whitespace()
142					.map(|s| s.to_lowercase())
143					.collect();
144
145				languages.append(&mut other_codes);
146			} else {
147				let code = available_languages[i].1.to_string();
148				languages.push(code);
149			}
150		}
151	}
152
153	// get content rating
154	let content_rating = match content_rating {
155		Some(rating) => rating,
156		None => {
157			let rating_options = &["Safe", "Contains NSFW content", "Primarily NSFW content"];
158			let selection = Select::new()
159				.with_prompt("Select content rating")
160				.items(rating_options)
161				.default(0)
162				.interact()?;
163
164			match selection {
165				0 => SourceContentRating::Safe,
166				1 => SourceContentRating::ContainsNsfw,
167				2 => SourceContentRating::PrimarilyNsfw,
168				_ => unreachable!(),
169			}
170		}
171	};
172
173	// create and move to the target directory if necessary
174	if target_path != current_dir {
175		std::fs::create_dir_all(&target_path).context("Failed to create new directory")?;
176		std::env::set_current_dir(&target_path).context("Failed to change directory")?;
177	}
178
179	// determine info from configuration
180	fn create_package_name(name: &str, template: bool) -> String {
181		let name = name
182			.chars()
183			.map(|c| c.to_ascii_lowercase()) // lowercase
184			.map(|c| if c == ' ' { '-' } else { c }) // replace spaces with hyphens
185			.filter(|c| c.is_ascii_alphanumeric() || *c == '-' || *c == '_') // ensure unicode-xid compliant
186			.collect::<String>()
187			.trim_matches(|c| c == '_' || c == '-') // remove trailing and leading underscores and hyphens
188			.to_string();
189		if name == "test" {
190			if template {
191				"test_template".to_string()
192			} else {
193				"test_source".to_string()
194			}
195		} else {
196			name
197		}
198	}
199	let mut package_name = create_package_name(&name, false);
200	// handle edge case for reserved package names
201	if package_name == "test" {
202		package_name = "test_source".to_string();
203	}
204	let id = format!(
205		"{}.{}",
206		if languages.len() == 1 {
207			&languages[0]
208		} else {
209			"multi"
210		},
211		package_name
212	);
213	let source_json = SourceJson {
214		info: SourceInfo {
215			id,
216			name,
217			alt_names: None,
218			version: 1,
219			url: Some(url),
220			urls: None,
221			content_rating: Some(content_rating),
222			languages,
223		},
224	};
225
226	if let Some(template_name) = template_name {
227		let template_package_name = create_package_name(&template_name, false);
228		let template_name = template_name
229			.chars()
230			.filter(|c| c.is_ascii_alphanumeric())
231			.collect::<String>();
232
233		// create workspace Cargo.toml
234		let cargo_toml = PathBuf::from("Cargo.toml");
235		std::fs::write(&cargo_toml, CARGO_TEMPLATE)
236			.context("Failed to create workspace Cargo.toml")?;
237
238		create_template_files(&template_package_name, &template_name)?;
239		create_source_files(
240			&package_name,
241			source_json,
242			Some((&template_package_name, &template_name)),
243		)?;
244	} else {
245		create_source_files(&package_name, source_json, None)?;
246		println!("configured a source");
247	}
248
249	Ok(())
250}

252// create a new cargo library and add aidoku dependencies to it
253// moves the current directory into a newly created directory if in_workspace is true
254fn create_empty_aidoku_lib(
255	name: &str,
256	in_workspace: bool,
257	template_name: Option<(&str, &str)>,
258) -> Result<(), anyhow::Error> {
259	// initialize via cargo
260	let mut cmd = &mut std::process::Command::new("cargo");
261	if in_workspace {
262		// if we're in a workspace, create a new directory
263		cmd = cmd.arg("new").arg(name);
264	} else {
265		// otherwise, use the current directory
266		cmd = cmd.arg("init").arg("--name").arg(name);
267	}
268	cmd.arg("--lib")
269		.stdout(std::process::Stdio::null())
270		.stderr(std::process::Stdio::null())
271		.status()
272		.context("Failed to initialize via cargo")?;
273
274	// if this is a workspace, move into the new directory we created
275	if in_workspace {
276		std::env::set_current_dir(PathBuf::from(name)).context("Failed to change directory")?;
277	}
278
279	// add aidoku library dependencies
280	let mut add_lib_cmd = std::process::Command::new("cargo");
281	let mut add_test_lib_cmd = std::process::Command::new("cargo");
282	let mut add_test_macro_cmd = std::process::Command::new("cargo");
283
284	let mut commands = vec![
285		// todo: remove --branch when this lands on main
286		add_lib_cmd
287			.arg("add")
288			.arg("aidoku")
289			.arg("--git")
290			.arg("https://github.com/Aidoku/aidoku-rs.git"),
291		add_test_lib_cmd
292			.arg("add")
293			.arg("aidoku")
294			.arg("--git")
295			.arg("https://github.com/Aidoku/aidoku-rs.git")
296			.arg("--features")
297			.arg("test")
298			.arg("--dev"),
299		add_test_macro_cmd
300			.arg("add")
301			.arg("aidoku-test")
302			.arg("--git")
303			.arg("https://github.com/Aidoku/aidoku-rs.git")
304			.arg("--dev"),
305	];
306
307	// if a template name is provided, add it as a dependency
308	let mut add_template_cmd = std::process::Command::new("cargo");
309	if let Some((template_package_name, _)) = template_name {
310		commands.push(
311			add_template_cmd
312				.arg("add")
313				.arg("--path")
314				.arg(format!("../{template_package_name}")),
315		);
316	}
317
318	for command in commands {
319		let result = command
320			.stdout(std::process::Stdio::null())
321			.stderr(std::process::Stdio::null())
322			.status()
323			.context("Failed to initialize source via cargo")?;
324
325		if !result.success() {
326			return Err(anyhow!("Failed to initialize source via cargo"));
327		}
328	}
329
330	Ok(())
331}

333// creates files for a new source
334fn create_source_files(
335	package_name: &str,
336	source_json: SourceJson,
337	template_name: Option<(&str, &str)>,
338) -> Result<(), anyhow::Error> {
339	let in_workspace = template_name.is_some();
340
341	// creates and moves into our new project folder if necessary
342	create_empty_aidoku_lib(package_name, in_workspace, template_name)?;
343
344	// override lib.rs
345	let lib_rs_path = PathBuf::from("src/lib.rs");
346	let source_name = source_json
347		.info
348		.name
349		.chars()
350		.filter(|c| c.is_ascii_alphanumeric())
351		.collect::<String>();
352	let lib_rs_content = if let Some((template_package_name, template_name)) = template_name {
353		SOURCE_TEMPLATE_LIB_TEMPLATE
354			.replace("{{SOURCE_NAME}}", &source_name)
355			.replace("{{TEMPLATE_NAME}}", template_name)
356			.replace(
357				"{{TEMPLATE_LIB_NAME}}",
358				&template_package_name.replace("-", "_"),
359			)
360	} else {
361		SOURCE_LIB_TEMPLATE.replace("{{SOURCE_NAME}}", &source_name)
362	};
363	std::fs::write(&lib_rs_path, lib_rs_content).context("Failed to write template lib.rs")?;
364
365	// add wasm config stuff on to Cargo.toml
366	let cargo_toml = PathBuf::from("Cargo.toml");
367	let mut cargo_toml_content =
368		std::fs::read_to_string(&cargo_toml).context("Failed to read Cargo.toml")?;
369	cargo_toml_content.push_str(CDYLIB_STR);
370	// if we're not in a workspace, add the profile stuff
371	if !in_workspace {
372		// use the profile configuration from workspace Cargo.toml template
373		let profile_text = CARGO_TEMPLATE
374			.lines()
375			.skip(3)
376			.collect::<Vec<_>>()
377			.join("\n");
378		cargo_toml_content.push_str(&profile_text);
379	}
380	std::fs::write(&cargo_toml, cargo_toml_content).context("Failed to write Cargo.toml")?;
381
382	// create .cargo/config.toml
383	let config_path = PathBuf::from(".cargo");
384	std::fs::create_dir_all(&config_path).context("Failed to create .cargo directory")?;
385	let config_toml = config_path.join("config.toml");
386	std::fs::write(&config_toml, CONFIG_TEMPLATE).context("Failed to create config.toml")?;
387
388	// create res folder
389	let res_path = PathBuf::from("res");
390	std::fs::create_dir_all(&res_path).context("Failed to create res directory")?;
391
392	// create source.json
393	let source_json_path = res_path.join("source.json");
394	let tabs_pretty = serde_json::ser::PrettyFormatter::with_indent(b"\t"); // indent with tabs
395	let file = std::fs::File::create(&source_json_path).context("Failed to create json file")?;
396	let mut w = std::io::BufWriter::new(file);
397	let mut ser = serde_json::Serializer::with_formatter(&mut w, tabs_pretty);
398	serde::Serialize::serialize(&source_json, &mut ser).context("Failed to write to json file")?;
399	w.write_all(b"\n")?; // add a newline after the json data
400	w.flush()?;
401
402	// create empty icon.png
403	let icon_path = res_path.join("icon.png");
404	std::fs::File::create(&icon_path).context("Failed to create empty icon.png")?;
405
406	// if we created and moved into a directory, move out of it
407	if in_workspace {
408		std::env::set_current_dir(PathBuf::from("..")).context("Failed to change directory")?;
409	}
410
411	Ok(())
412}

414// creates files for a new source template
415fn create_template_files(package_name: &str, name: &str) -> Result<(), anyhow::Error> {
416	create_empty_aidoku_lib(package_name, true, None)?;
417
418	// override lib.rs
419	let lib_rs_path = PathBuf::from("src/lib.rs");
420	let lib_rs_content = TEMPLATE_LIB_TEMPLATE.replace("{{TEMPLATE_NAME}}", name);
421	std::fs::write(&lib_rs_path, lib_rs_content).context("Failed to write lib.rs template")?;
422
423	// move out of the directory we created
424	std::env::set_current_dir(PathBuf::from("..")).context("Failed to change directory")?;
425
426	Ok(())
