1//! Build and package an Aidoku source.
2use anyhow::{anyhow, Context};
3use std::io::prelude::*;
4
5pub fn run(path: Option<std::path::PathBuf>) -> anyhow::Result<()> {
6	// move to the directory if specified
7	if let Some(path) = path {
8		let path = path.as_path();
9		if !path.exists() {
10			return Err(anyhow!("Path does not exist"));
11		}
12		std::env::set_current_dir(path).context("Failed to change directory")?;
13	}
14
15	let current_path = std::env::current_dir().context("Failed to get current directory")?;
16
17	// build rust source
18	let result = std::process::Command::new("cargo")
19		.arg("build")
20		.arg("--release")
21		.arg("--target")
22		.arg("wasm32-unknown-unknown")
23		.status()
24		.context("Failed to build source")?;
25	if !result.success() {
26		return Err(anyhow!("Build failed"));
27	}
28
29	// folder containing `source.json` file
30	let res_dir = current_path.join("res");
31
32	if !res_dir.exists() {
33		return Err(anyhow!("res directory does not exist"));
34	}
35
36	// find the build folder containing resulting wasm file
37	// start from current directory and go up to 3 parent directories if not found
38	let build_dir = {
39		let mut result = None;
40
41		let mut cur_parent = current_path.clone();
42		let mut build_dir: std::path::PathBuf;
43
44		let max_parents = 3; // only check up to 3 parent directories
45		let mut parent_count = 0;
46
47		loop {
48			build_dir = cur_parent
49				.join("target")
50				.join("wasm32-unknown-unknown")
51				.join("release");
52
53			if build_dir.exists() {
54				result = Some(build_dir);
55				break;
56			}
57
58			if let Some(parent) = cur_parent.parent() {
59				cur_parent = parent.to_path_buf();
60				parent_count += 1;
61				if parent_count >= max_parents {
62					// reached the max parent directories to check
63					break;
64				}
65			} else {
66				// reached the root directory, no parent directories left to check
67				break;
68			}
69		}
70
71		result
72	};
73	let Some(build_dir) = build_dir else {
74		return Err(anyhow!("Failed to find build target directory"));
75	};
76
77	// folder containing source data we will create and zip
78	let payload_dir = build_dir.join("Payload");
79
80	// create payload directory
81	std::fs::create_dir_all(&payload_dir).context("Failed to create package directory")?;
82
83	// copy res files to payload directory
84	for entry in std::fs::read_dir(res_dir).context("Failed to read res directory")? {
85		let path = entry
86			.context("Failed to read entry in res directory")?
87			.path();
88
89		if path.is_file() {
90			let file_name = path.file_name().expect("Valid file must have a name");
91			let dest_path = payload_dir.join(file_name);
92
93			std::fs::copy(&path, &dest_path).context("Failed to copy file from res directory")?;
94		}
95	}
96
97	// copy wasm file to package directory
98	// note that this will just copy the first wasm file it finds in the build dir
99	// if the cargo package name changes, this might copy the wrong (old) file
100	let mut found_binary = false;
101	for entry in std::fs::read_dir(build_dir).context("Failed to read build directory")? {
102		let entry = entry.context("Failed to read entry in build directory")?;
103		let path = entry.path();
104
105		match path.extension() {
106			Some(ext) if ext == "wasm" => {
107				let dest_path = payload_dir.join("main.wasm");
108				std::fs::copy(&path, &dest_path)
109					.context("Failed to copy file from build directory")?;
110				found_binary = true;
111				break;
112			}
113			_ => {}
114		}
115	}
116	if !found_binary {
117		return Err(anyhow!("No wasm file found in build directory"));
118	}
119
120	// zip payload directory
121	let zip_output = current_path.join("package.aix");
122	create_zip(payload_dir.as_path(), zip_output.as_path())
123		.context("Failed to compress payload directory")?;
124
125	// remove package dir
126	std::fs::remove_dir_all(payload_dir).context("Failed to remove payload directory")?;
127
128	Ok(())
129}

131// zip a directory and output to a file
132fn create_zip(src_dir: &std::path::Path, dst_file: &std::path::Path) -> std::io::Result<()> {
133	let file = std::fs::File::create(dst_file)?;
134	let mut zip = zip::ZipWriter::new(file);
135	let options = zip::write::SimpleFileOptions::default()
136		.compression_method(zip::CompressionMethod::Deflated)
137		.unix_permissions(0o755);
138
139	let dir_name = src_dir
140		.file_name()
141		.and_then(|n| n.to_str())
142		.unwrap_or("Payload");
143
144	for entry in std::fs::read_dir(src_dir)? {
145		let entry = entry?;
146		let path = entry.path();
147
148		if path.is_file() {
149			// let relative_path = path.strip_prefix(src_dir_str).unwrap().to_str().unwrap();
150			let Some(file_name) = path.file_name().and_then(|n| n.to_str()) else {
151				continue;
152			};
153			let zip_path = format!("{}/{}", dir_name, file_name);
154
155			zip.start_file(zip_path, options)?;
156			let mut f = std::fs::File::open(path)?;
157			let mut buffer = Vec::new();
158			f.read_to_end(&mut buffer)?;
159			zip.write_all(&buffer)?;
160		}
161	}
162
163	zip.finish()?;
164	Ok(())
165}
