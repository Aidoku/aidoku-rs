1//! Serve an Aidoku source as a local source list.
2use axum::{
3	http::{header::USER_AGENT, Request, Response},
4	Router,
5};
6use std::{net::SocketAddr, time::Duration};
7use tower_http::{services::ServeDir, trace::TraceLayer};
8use tracing::Level;
9
10pub async fn run(
11	files: Vec<std::path::PathBuf>,
12	output_path: &std::path::PathBuf,
13	port: u16,
14) -> anyhow::Result<()> {
15	super::build::run(files, output_path, Some("Development Source List".into()))?;
16
17	// enable logging
18	tracing_subscriber::fmt()
19		.with_target(false) // hide the `aidoku_cli::commands::serve:` text
20		.with_max_level(Level::INFO)
21		.init();
22
23	// register static router for serving the output directory
24	let app = Router::new()
25		.fallback_service(ServeDir::new(output_path))
26		.layer(
27			TraceLayer::new_for_http()
28				.make_span_with(|req: &Request<axum::body::Body>| {
29					// show the request user agent in the log
30					let user_agent = req
31						.headers()
32						.get(USER_AGENT)
33						.and_then(|h| h.to_str().ok())
34						.unwrap_or("unknown");
35
36					tracing::info_span!(
37						"request",
38						method = %req.method(),
39						uri = %req.uri(),
40						user_agent = %user_agent,
41					)
42				})
43				.on_response(
44					|response: &Response<axum::body::Body>,
45					 _latency: Duration,
46					 span: &tracing::Span| {
47						// log with error level if the http response is an error
48						let status = response.status();
49						if status.is_server_error() || status.is_client_error() {
50							span.in_scope(|| {
51								tracing::error!(
52									status = %status,
53								);
54							});
55						} else {
56							span.in_scope(|| {
57								tracing::info!("");
58							});
59						};
60					},
61				),
62		);
63
64	let addr = SocketAddr::from(([0, 0, 0, 0], port));
65
66	let local_ip_address = local_ip_address::local_ip()
67		.map(|ip| ip.to_string())
68		.unwrap_or("localhost".into());
69
70	println!(
71		"Serving source list at http://{}:{}",
72		local_ip_address, port
73	);
74	println!(
75		"Add http://{}:{}/index.min.json as a source list in Aidoku",
76		local_ip_address, port
77	);
78	println!("Hit CTRL-C to stop the server.");
79
80	// start http server
81	axum::serve(tokio::net::TcpListener::bind(addr).await?, app)
82		.await
83		.map_err(anyhow::Error::from)
84}
