8pub fn print(mut env: FunctionEnvMut<WasmEnv>, ptr: Ptr, len: u32) {
9	let Ok(str) = env.data().read_string(&env, ptr, len) else {
10		env.data_mut()
11			.write_stdout("error: failed to read string for printing.\n");
12		return;
13	};
14	env.data_mut().write_stdout(&str);
15	env.data_mut().write_stdout("\n");
16}
