6enum Result {
7	Success,
8	InvalidDescriptor,
9	// InvalidBufferSize,
10	FailedMemoryWrite,
11	InvalidString,
12	InvalidDateString,
13}

15impl From<Result> for i32 {
16	fn from(result: Result) -> i32 {
17		match result {
18			Result::Success => 0,
19			Result::InvalidDescriptor => -1,
20			// Result::InvalidBufferSize => -2,
21			Result::FailedMemoryWrite => -3,
22			Result::InvalidString => -4,
23			Result::InvalidDateString => -5,
24		}
25	}
26}

32pub fn buffer_len(env: FunctionEnvMut<WasmEnv>, rid: Rid) -> FFIResult {
33	let Some(data) = env.data().store.get(rid).and_then(|item| item.as_encoded()) else {
34		return Result::InvalidDescriptor.into();
35	};
36	data.len() as FFIResult
37}

39pub fn read_buffer(env: FunctionEnvMut<WasmEnv>, rid: Rid, ptr: Ptr, size: u32) -> FFIResult {
40	let Some(data) = env.data().store.get(rid).and_then(|item| item.as_encoded()) else {
41		return Result::InvalidDescriptor.into();
42	};
43	if size as usize <= data.len() {
44		let data = data.iter().take(size as usize).copied().collect::<Vec<_>>();
45		if env.data().write_buffer(&env, ptr, &data).is_err() {
46			Result::FailedMemoryWrite.into()
47		} else {
48			Result::Success.into()
49		}
50	} else {
51		Result::FailedMemoryWrite.into()
52	}
53}

65pub fn parse_date(
66	env: FunctionEnvMut<WasmEnv>,
67	date_str: Ptr,
68	date_len: u32,
69	format_str: Ptr,
70	format_len: u32,
71	_locale_str: Ptr,
72	_locale_len: u32,
73	timezone_str: Ptr,
74	timezone_len: u32,
75) -> f64 {
76	let Ok(string) = env.data().read_string(&env, date_str, date_len) else {
77		return Into::<i32>::into(Result::InvalidString) as f64;
78	};
79	let Ok(format) = env.data().read_string(&env, format_str, format_len) else {
80		return Into::<i32>::into(Result::InvalidString) as f64;
81	};
82	let timezone_string = if timezone_len > 0 {
83		env.data()
84			.read_string(&env, timezone_str, timezone_len)
85			.ok()
86	} else {
87		None
88	};
89
90	let timezone: chrono_tz::Tz = timezone_string
91		.as_deref()
92		.and_then(|tz_str| tz_str.parse().ok())
93		.unwrap_or(chrono_tz::UTC);
94
95	use chrono::NaiveDateTime;
96
97	let Some(timestamp) =
98		NaiveDateTime::parse_from_str(&string, &swift_date_format_to_chrono(&format))
99			.ok()
100			.and_then(|dt| dt.and_local_timezone(timezone).single())
101			.map(|dt| dt.timestamp() as f64)
102	else {
103		return Into::<i32>::into(Result::InvalidDateString) as f64;
104	};
105	timestamp
106}

108fn swift_date_format_to_chrono(format: &str) -> String {
109	let mut result = String::new();
110	let mut chars = format.chars().peekable();
111
112	while let Some(c) = chars.next() {
113		let token = match c {
114			// year
115			'y' => {
116				// yyyy or yy
117				let mut count = 1;
118				while chars.peek() == Some(&'y') {
119					chars.next();
120					count += 1;
121				}
122				if count == 2 { "%y" } else { "%Y" }
123			}
124			// month
125			'M' => {
126				// MM or MMM or MMMM
127				let mut count = 1;
128				while chars.peek() == Some(&'M') {
129					chars.next();
130					count += 1;
131				}
132				match count {
133					1 | 2 => "%m",
134					3 => "%b",
135					_ => "%B",
136				}
137			}
138			// day
139			'd' => {
140				// dd or d
141				while chars.peek() == Some(&'d') {
142					chars.next();
143				}
144				"%d"
145			}
146			// hour
147			'H' => {
148				// HH or H
149				while chars.peek() == Some(&'H') {
150					chars.next();
151				}
152				"%H"
153			}
154			'h' => {
155				// hh or h
156				while chars.peek() == Some(&'h') {
157					chars.next();
158				}
159				"%I"
160			}
161			// minute
162			'm' => {
163				// mm or m
164				while chars.peek() == Some(&'m') {
165					chars.next();
166				}
167				"%M"
168			}
169			// second
170			's' => {
171				// ss or s
172				while chars.peek() == Some(&'s') {
173					chars.next();
174				}
175				"%S"
176			}
177			'a' => "%p", // AM/PM
178			// weekday
179			'E' => {
180				// EEE or EEEE
181				let mut count = 1;
182				while chars.peek() == Some(&'E') {
183					chars.next();
184					count += 1;
185				}
186				if count >= 4 { "%A" } else { "%a" }
187			}
188			// timezone
189			'z' => {
190				let mut count = 1;
191				while chars.peek() == Some(&c) {
192					chars.next();
193					count += 1;
194				}
195				if count >= 4 { "%Z" } else { "%z" }
196			}
197			'Z' => {
198				while chars.peek() == Some(&c) {
199					chars.next();
200				}
201				"%Z"
202			}
203			// literal
204			c => {
205				result.push(c);
206				continue;
207			}
208		};
209		result.push_str(token);
210	}
211	result
212}
