1//! Parses date strings into Unix timestamps.
2use chrono::{Duration, NaiveDate, TimeZone, Utc};
3
4static mut UTC_OFFSET: i64 = -1;
5
6#[link(wasm_import_module = "std")]
7extern "C" {
8	fn utc_offset() -> i64;
9}
10
11unsafe fn get_utc_offset() -> i64 {
12	if UTC_OFFSET == -1 {
13		UTC_OFFSET = utc_offset();
14	}
15	UTC_OFFSET
16}
17
18pub fn parse_date<T: AsRef<str>, U: AsRef<str>>(date_str: T, format: U) -> Option<i64> {
19	NaiveDate::parse_from_str(date_str.as_ref(), format.as_ref())
20		.ok()?
21		.and_hms_opt(0, 0, 0)
22		.map(|d| Utc.from_utc_datetime(&d).timestamp())
23}
24
25pub fn parse_local_date<T: AsRef<str>, U: AsRef<str>>(date_str: T, format: U) -> Option<i64> {
26	let date = NaiveDate::parse_from_str(date_str.as_ref(), format.as_ref())
27		.ok()?
28		.and_hms_opt(0, 0, 0)?;
29	let offset = unsafe { get_utc_offset() };
30	let adjusted = date + Duration::seconds(offset);
31	Some(Utc.from_utc_datetime(&adjusted).timestamp())
32}
