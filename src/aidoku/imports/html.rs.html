65/// Error type for HTML operations.
66#[derive(PartialEq, Eq, Debug, Clone)]
67pub enum HtmlError {
68	InvalidDescriptor,
69	InvalidString,
70	InvalidHtml,
71	InvalidQuery,
72	NoResult,
73	SwiftSoupError,
74}
75
76impl HtmlError {
77	fn from(value: FFIResult) -> Option<Self> {
78		match value {
79			-1 => Some(Self::InvalidDescriptor),
80			-2 => Some(Self::InvalidString),
81			-3 => Some(Self::InvalidHtml),
82			-4 => Some(Self::InvalidQuery),
83			-5 => Some(Self::NoResult),
84			-6 => Some(Self::SwiftSoupError),
85			_ => None,
86		}
87	}
88}

95	/// Parse HTML into a Document.
96	///
97	/// As there is no base URL specified, absolute URL resolution requires the
98	/// HTML to have a `<base href>` tag.
99	pub fn parse<T: AsRef<[u8]>>(html: T) -> Result<Document, HtmlError> {
100		let buf = html.as_ref();
101		let rid = unsafe { parse(buf.as_ptr(), buf.len(), "".as_ptr(), 0) };
102		if let Some(error) = HtmlError::from(rid) {
103			Err(error)
104		} else {
105			Ok(Document(unsafe { Element::from(rid) }))
106		}
107	}
108
109	/// Parse HTML into a Document, with a base URL.
110	///
111	/// The given `base_url` will be used for any URLs that occurs before a
112	/// `<base href>` tag is defined.
113	pub fn parse_with_url<T: AsRef<[u8]>, B: AsRef<str>>(
114		html: T,
115		base_url: B,
116	) -> Result<Document, HtmlError> {
117		let buf = html.as_ref();
118		let url = base_url.as_ref();
119		let rid = unsafe { parse(buf.as_ptr(), buf.len(), url.as_ptr(), url.len()) };
120		if let Some(error) = HtmlError::from(rid) {
121			Err(error)
122		} else {
123			Ok(Document(unsafe { Element::from(rid) }))
124		}
125	}

127	/// Parse a HTML fragment, assuming that it forms the `body` of the HTML.
128	///
129	/// Similar to [Html::parse], relative URLs will not be resolved unless
130	/// there is a `<base href>` tag.
131	pub fn parse_fragment<T: AsRef<[u8]>>(html: T) -> Result<Document, HtmlError> {
132		let buf = html.as_ref();

134		if let Some(error) = HtmlError::from(rid) {

138		}
139	}

145	pub fn parse_fragment_with_url<T: AsRef<[u8]>, B: AsRef<str>>(

159	/// Escape any HTML-reserved characters to HTML entities.
160	///
161	/// # Examples
162	/// ```ignore
163	/// use aidoku::imports::html::Html;
164	/// assert_eq!(
165	///     Html::escape("Hello &<> Å å π 新 there ¾ © »"),
166	///     "Hello &amp;&lt;&gt; Å å π 新 there ¾ © »",
167	/// );
168	/// ```
169	pub fn escape<T: AsRef<str>>(text: T) -> String {
170		let text = text.as_ref();
171		let rid = unsafe { escape(text.as_ptr(), text.len()) };
172		read_string_and_destroy(rid).unwrap_or_default()
173	}
174
175	/// Unescape any HTML entities to their original characters.
176	///
177	/// # Examples
178	/// ```ignore
179	/// use aidoku::imports::html::Html;
180	/// assert_eq!(
181	///     Html::unescape("Hello &amp;&lt;&gt; Å å π 新 there ¾ © »"),
182	///     Some("Hello &<> Å å π 新 there ¾ © »".into()),
183	/// );
184	/// ```
185	pub fn unescape<T: AsRef<str>>(text: T) -> Option<String> {
186		let text = text.as_ref();
187		let rid = unsafe { unescape(text.as_ptr(), text.len()) };
188		if HtmlError::from(rid).is_some() {
189			return None;
190		}
191		read_string_and_destroy(rid)
192	}

204	/// Find elements that match the given CSS (or JQuery) selector.
205	///
206	/// <details>
207	///     <summary>Supported selectors</summary>
208	///
209	/// | Pattern                 | Matches                                                                                              | Example                                                           |
210	/// |-------------------------|------------------------------------------------------------------------------------------------------|-------------------------------------------------------------------|
211	/// | `*`                     | any element                                                                                          | `*`                                                               |
212	/// | `tag`                   | elements with the given tag name                                                                     | `div`                                                             |
213	/// | <code>*\|E</code>       | elements of type E in any namespace (including non-namespaced)                                       | <code>*\|name</code> finds `<fb:name>` and `<name>` elements      |
214	/// | <code>ns\|E</code>      | elements of type E in the namespace ns                                                               | <code>fb\|name</code> finds `<fb:name>` elements                  |
215	/// | `#id`                   | elements with attribute ID of "id"                                                                   | `div#wrap`, `#logo`                                               |
216	/// | `.class`                | elements with a class name of "class"                                                                | `div.left`, `.result`                                             |
217	/// | `[attr]`                | elements with an attribute named "attr" (with any value)                                             | `a[href]`, `[title]`                                              |
218	/// | `[^attrPrefix]`         | elements with an attribute name starting with "attrPrefix". Use to find elements with HTML5 datasets | `[^data-]`, `div[^data-]`                                         |
219	/// | `[attr=val]`            | elements with an attribute named "attr", and value equal to "val"                                    | `img[width=500]`, `a[rel=nofollow]`                               |
220	/// | `[attr="val"]`          | elements with an attribute named "attr", and value equal to "val"                                    | `span[hello="Cleveland"][goodbye="Columbus"]`, `a[rel="nofollow"]`|
221	/// | `[attr^=valPrefix]`     | elements with an attribute named "attr", and value starting with "valPrefix"                         | `a[href^=http:]`                                                  |
222	/// | `[attr$=valSuffix]`     | elements with an attribute named "attr", and value ending with "valSuffix"                           | `img[src$=.png]`                                                  |
223	/// | `[attr*=valContaining]` | elements with an attribute named "attr", and value containing "valContaining"                        | `a[href*=/search/]`                                               |
224	/// | `[attr~=regex]`         | elements with an attribute named "attr", and value matching the regular expression                   | `img[src~=(?i)\\.(png\|jpe?g)]`                                   |
225	/// |                         | The above may be combined in any order                                                               | `div.header[title]`                                               |
226	///
227	/// ## Combinators
228	/// | Pattern   | Matches                                         | Example                     |
229	/// |-----------|-------------------------------------------------|-----------------------------|
230	/// | `E F`     | an F element descended from an E element        | `div a`, `.logo h1`         |
231	/// | `E > F`   | an F direct child of E                          | `ol > li`                   |
232	/// | `E + F`   | an F element immediately preceded by sibling E  | `li + li`, `div.head + div` |
233	/// | `E ~ F`   | an F element preceded by sibling E              | `h1 ~ p`                    |
234	/// | `E, F, G` | all matching elements E, F, or G                | `a[href], div, h3`          |
235	///
236	/// ## Pseudo selectors
237	/// | Pattern              | Matches                                                                                                                                                   | Example                                                                                                                                                      |
238	/// |----------------------|-----------------------------------------------------------------------------------------------------------------------------------------------------------|--------------------------------------------------------------------------------------------------------------------------------------------------------------|
239	/// | `:lt(n)`             | elements whose sibling index is less than n                                                                                                               | `td:lt(3)` finds the first 3 cells of each row                                                                                                               |
240	/// | `:gt(n)`             | elements whose sibling index is greater than n                                                                                                            | `td:gt(1)` finds cells after skipping the first two                                                                                                          |
241	/// | `:eq(n)`             | elements whose sibling index is equal to n                                                                                                                | `td:eq(0)` finds the first cell of each row                                                                                                                  |
242	/// | `:has(selector)`     | elements that contains at least one element matching the selector                                                                                         | `div:has(p)` finds divs that contain p elements; `div:has(> a)` selects div elements that have at least one direct child a element.                          |
243	/// | `:not(selector)`     | elements that do not match the selector.                                                                                                                  | `div:not(.logo)` finds all divs that do not have the "logo" class; `div:not(:has(div))` finds divs that do not contain divs.                                 |
244	/// | `:contains(text)`    | elements that contains the specified text. The search is case insensitive. The text may appear in the found element, or any of its descendants.           | `p:contains(SwiftSoup)` finds p elements containing the text "SwiftSoup"; `p:contains(hello \(there\))` finds p elements containing the text "Hello (There)" |
245	/// | `:matches(regex)`    | elements whose text matches the specified regular expression. The text may appear in the found element, or any of its descendants.                        | `td:matches(\\d+)` finds table cells containing digits. div:matches((?i)login) finds divs containing the text, case insensitively.                           |
246	/// | `:containsOwn(text)` | elements that directly contain the specified text. The search is case insensitive. The text must appear in the found element, not any of its descendants. | `p:containsOwn(SwiftSoup)` finds p elements with own text "SwiftSoup".                                                                                       |
247	/// | `:matchesOwn(regex)` | elements whose own text matches the specified regular expression. The text must appear in the found element, not any of its descendants.                  | `td:matchesOwn(\\d+)` finds table cells directly containing digits. div:matchesOwn((?i)login) finds divs containing the text, case insensitively.            |
248	///
249	/// ## Structural pseudo-selectors
250	/// | Pattern                   | Matches                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                           | Example                                                |
251	/// |---------------------------|-----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------|--------------------------------------------------------|
252	/// | `:root`                   | The element that is the root of the document. In HTML, this is the html element                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                   |                                                        |                                                                                                                                                                                                 |
253	/// | `:nth-child(an+b)`        | elements that have an+b-1 siblings before it in the document tree, for any positive integer or zero value of n, and has a parent element. For values of a and b greater than zero, this effectively divides the element's children into groups of a elements (the last group taking the remainder), and selecting the bth element of each group. For example, this allows the selectors to address every other row in a table, and could be used to alternate the color of paragraph text in a cycle of four. The a and b values must be integers (positive, negative, or zero). The index of the first child of an element is 1. |                                                        |
254	/// | `:nth-last-child(an+b)`   | elements that have an+b-1 siblings after it in the document tree. Otherwise like `:nth-child()`                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                   | `tr:nth-last-child(-n+2)` the last two rows of a table |
255	/// | `:nth-of-type(an+b)`      | pseudo-class notation represents an element that has an+b-1 siblings with the same expanded element name before it in the document tree, for any zero or positive integer value of n, and has a parent element                                                                                                                                                                                                                                                                                                                                                                                                                    | `img:nth-of-type(2n+1)`                                |
256	/// | `:nth-last-of-type(an+b)` | pseudo-class notation represents an element that has an+b-1 siblings with the same expanded element name after it in the document tree, for any zero or positive integer value of n, and has a parent element                                                                                                                                                                                                                                                                                                                                                                                                                     | `img:nth-last-of-type(2n+1)`                           |
257	/// | `:first-child`            | elements that are the first child of some other element.                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                          | `div > p:first-child`                                  |
258	/// | `:last-child`             | elements that are the last child of some other element.                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                           | `ol > li:last-child`                                   |
259	/// | `:first-of-type`          | elements that are the first sibling of its type in the list of children of its parent element                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                     | `dl dt:first-of-type`                                  |
260	/// | `:last-of-type`           | elements that are the last sibling of its type in the list of children of its parent element                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                      | `tr > td:last-of-type`                                 |
261	/// | `:only-child`             | elements that have a parent element and whose parent element hasve no other element children                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                      |                                                        |
262	/// | `:only-of-type`           |  an element that has a parent element and whose parent element has no other element children with the same expanded element name                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                  |                                                        |
263	/// | `:empty`                  | elements that have no children at all                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                             |                                                        |
264	/// </details>
265	pub fn select<T: AsRef<str>>(&self, css_query: T) -> Option<ElementList> {
266		self.0.select(css_query)
267	}

473	/// The given HTML will be parsed, and each node prepended to the start

590/// A collection of HTML elements.

598impl ElementList {
599	/// Get an instance from a [Rid].
600	unsafe fn from(rid: Rid) -> Self {
601		let size = size(rid) as usize;
602		Self {
603			rid,
604			lower_bound: 0,
605			upper_bound: size.wrapping_sub(1),
606			size,
607		}
608	}

712impl Iterator for ElementList {
713	type Item = Element;
714
715	fn next(&mut self) -> Option<Self::Item> {
716		if self.lower_bound > self.upper_bound || self.upper_bound == usize::MAX {
717			return None;
718		}
719		let value_ref = self.get(self.lower_bound);
720		self.lower_bound += 1;
721		value_ref
722	}
723}
724
725impl DoubleEndedIterator for ElementList {
726	fn next_back(&mut self) -> Option<Self::Item> {
727		if self.lower_bound > self.upper_bound || self.upper_bound == usize::MAX {
728			return None;
729		}
730		let value_ref = self.get(self.upper_bound);
731		self.upper_bound = self.upper_bound.wrapping_sub(1);
732		value_ref
733	}
734}

738		unsafe { destroy(self.rid) }
