66/// Error type for network requests.
67#[derive(PartialEq, Eq, Debug, Clone, Copy)]
68pub enum RequestError {
69	InvalidDescriptor,
70	InvalidString,
71	InvalidMethod,
72	InvalidUrl,
73	InvalidHtml,
74	InvalidBufferSize,
75	MissingData,
76	MissingResponse,
77	MissingUrl,
78	RequestError,
79	FailedMemoryWrite,
80	NotAnImage,
81	Closed,
82}
83
84impl RequestError {
85	fn from(value: i32) -> Option<Self> {
86		match value {
87			-1 => Some(Self::InvalidDescriptor),
88			-2 => Some(Self::InvalidString),
89			-3 => Some(Self::InvalidMethod),
90			-4 => Some(Self::InvalidUrl),
91			-5 => Some(Self::InvalidHtml),
92			-6 => Some(Self::InvalidBufferSize),
93			-7 => Some(Self::MissingData),
94			-8 => Some(Self::MissingResponse),
95			-9 => Some(Self::MissingUrl),
96			-10 => Some(Self::RequestError),
97			-11 => Some(Self::FailedMemoryWrite),
98			-12 => Some(Self::NotAnImage),
99			_ => None,
100		}
101	}
102}

144impl Request {
145	/// Create a new request with a URL and HTTP method.
146	///
147	/// Returns an error if the provided URL is invalid.
148	///
149	/// # Examples
150	///
151	/// ```ignore
152	/// use aidoku::imports::net::{HttpMethod, Request};
153	/// Request::new("https://example.com", HttpMethod::Get).unwrap();
154	/// ```
155	pub fn new<T: AsRef<str>>(url: T, http_method: HttpMethod) -> Result<Self, RequestError> {
156		let url = url.as_ref();
157		unsafe {
158			let rid = init(http_method);
159			let mut request = Self {
160				rid,
161				http_method,
162				url: None,
163				should_close: true,
164			};
165			request.set_url(url)?;
166			Ok(request)
167		}
168	}
169
170	convenience_http_methods! { get, HttpMethod::Get, "Create a new GET request with the given URL." }
171	convenience_http_methods! { post, HttpMethod::Post, "Create a new POST request with the given URL." }
172	convenience_http_methods! { put, HttpMethod::Put, "Create a new PUT request with the given URL." }
173	convenience_http_methods! { head, HttpMethod::Head, "Create a new HEAD request with the given URL." }
174	convenience_http_methods! { delete, HttpMethod::Delete, "Create a new DELETE request with the given URL." }

176	/// Send multiple requests in parallel, and wait for all of them to finish.
177	pub fn send_all<I>(requests: I) -> Vec<Result<Response, RequestError>>
178	where
179		I: IntoIterator<Item = Request>,
180	{
181		let mut ids: Vec<i32> = Vec::new();
182		// mark all requests as sent and add their IDs to the vector
183		let responses = requests
184			.into_iter()
185			.map(|mut r| {
186				r.should_close = false;
187				ids.push(r.rid);
188				Ok(Response::from(r))
189			})
190			.collect();
191
192		let result = unsafe { send_all(ids.as_mut_ptr(), ids.len()) };
193
194		if result == 0 {
195			// success, return result
196			responses
197		} else {
198			// one or more of the requests failed
199			// the error codes are stored in the ids vector
200			let mut idx = 0;
201			responses
202				.into_iter()
203				.map(|response| {
204					let error = ids.get(idx).and_then(|id| RequestError::from(*id));
205					let result = match error {
206						Some(e) => Err(e),
207						None => response,
208					};
209					idx += 1;
210					result
211				})
212				.collect()
213		}
214	}

224		let key = key.as_ref();

243	/// Set the URL for the request.
244	pub fn set_url<T: AsRef<str>>(&mut self, url: T) -> Result<(), RequestError> {
245		let url = url.as_ref();
246		self.url = Some(String::from(url));
247		let result = unsafe { set_url(self.rid, url.as_ptr(), url.len()) };
248		if let Some(error) = RequestError::from(result) {
249			Err(error)
250		} else {
251			Ok(())
252		}
253	}
254
255	/// Get the URL of the request.
256	pub fn url(&self) -> Option<&String> {
257		self.url.as_ref()
258	}

260	/// Send the request.
261	#[inline]
262	pub fn send(mut self) -> Result<Response, RequestError> {
263		let result = unsafe { send(self.rid) };
264		if let Some(error) = RequestError::from(result) {
265			Err(error)
266		} else {
267			self.should_close = false;
268			Ok(Response::from(self))
269		}
270	}

304impl Response {
305	/// Get the response's status code.
306	#[inline]
307	pub fn status_code(&self) -> i32 {
308		unsafe { get_status_code(self.rid) }
309	}
310
311	/// Get a response HTTP header.
312	pub fn get_header<T: AsRef<str>>(&self, header: T) -> Option<String> {
313		let header = header.as_ref();
314		let rid = unsafe { get_header(self.rid, header.as_ptr(), header.len()) };
315		if rid < 0 {
316			return None;
317		}
318		read_string_and_destroy(rid)
319	}
320
321	/// Get the raw data from the response.
322	pub fn get_data(&self) -> Result<Vec<u8>, RequestError> {
323		let size = unsafe { data_len(self.rid) };
324		if let Some(error) = RequestError::from(size) {
325			return Err(error);
326		}
327		let size = size as usize;
328		let mut buffer: Vec<u8> = Vec::with_capacity(size);
329		unsafe {
330			let result = read_data(self.rid, buffer.as_mut_ptr(), size);
331			if let Some(error) = RequestError::from(result) {
332				return Err(error);
333			}
334			buffer.set_len(size);
335		}
336		Ok(buffer)
337	}
338
339	/// Gets the response data as an image.
340	pub fn get_image(&self) -> Result<ImageRef, RequestError> {
341		let result = unsafe { get_image(self.rid) };
342		if let Some(error) = RequestError::from(result) {
343			Err(error)
344		} else {
345			Ok(ImageRef::from(result, false))
346		}
347	}
348
349	/// Gets the response data as a string.
350	pub fn get_string(&self) -> Result<String, AidokuError> {
351		let res = String::from_utf8(self.get_data()?);
352		match res {
353			Ok(res) => Ok(res),
354			Err(err) => Err(AidokuError::Utf8Error(err.utf8_error())),
355		}
356	}
357
358	/// Get the response data as an HTML [Document].
359	pub fn get_html(&self) -> Result<Document, RequestError> {
360		let rid = unsafe { html(self.rid) };
361		if let Some(error) = RequestError::from(rid) {
362			return Err(error);
363		}
364		Ok(unsafe { Document::from(rid) })
365	}

367	/// Create a new request with the same method and url as the one sent to get this response.
368	pub fn into_request(self) -> Request {
369		let rid = unsafe { init(self.http_method) };
370		if let Some(url) = self.url.as_ref() {
371			_ = unsafe { set_url(rid, url.as_ptr(), url.len()) };
372		}
373		Request {
374			rid: rid,
375			http_method: self.http_method,
376			url: self.url.clone(),
377			should_close: true,
378		}
379	}
380}

382#[cfg(feature = "json")]
383impl Response {
384	/// Get the response data as a JSON value. This requires the request to stay in scope so the data can be referenced.
385	pub fn get_json<'a, T>(&'a mut self) -> Result<T, AidokuError>
386	where
387		T: serde::de::Deserialize<'a>,
388	{
389		let data = self.get_data()?;
390		self.data = Some(data);
391		let value = serde_json::from_slice(self.data.as_ref().unwrap())
392			.map_err(|_| AidokuError::JsonParseError)?;
393		Ok(value)
394	}
395
396	/// Get the response data as an owned JSON value.
397	pub fn get_json_owned<T>(self) -> Result<T, AidokuError>
398	where
399		T: serde::de::DeserializeOwned,
400	{
401		let data = self.get_data()?;
402		let value = serde_json::from_slice(&data).map_err(|_| AidokuError::JsonParseError)?;
403		Ok(value)
404	}
405}
