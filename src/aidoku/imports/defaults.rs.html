13#[link(wasm_import_module = "defaults")]
14extern "C" {
15	fn get(key: *const u8, len: usize) -> FFIResult;
16	fn set(key: *const u8, len: usize, kind: u8, value: Ptr) -> FFIResult;

19/// A default value that can be stored in UserDefaults.
20pub enum DefaultValue {
21	Bool(bool),
22	Int(i32),
23	Float(f32),
24	String(String),
25	StringArray(Vec<String>),
26	Null,
27	HashMap(HashMap<String, String>),
28}
29
30impl DefaultValue {
31	fn as_byte(&self) -> u8 {
32		match self {
33			Self::Bool(_) => 1,
34			Self::Int(_) => 2,
35			Self::Float(_) => 3,
36			Self::String(_) => 4,
37			Self::StringArray(_) => 5,
38			Self::Null => 6,
39			Self::HashMap(_) => 0, // not a valid default value
40		}
41	}
42}
43
44/// Returns the UserDefaults value associated with the specified key.
45pub fn defaults_get<Value: DeserializeOwned>(key: &str) -> Option<Value> {
46	let rid = unsafe { get(key.as_ptr(), key.len()) };
47	if rid < 0 {
48		return None;
49	}
50	let result = read::<Value>(rid).ok();
51	unsafe { destroy(rid) };
52	result
53}
54
55/// Returns a HashMap stored in UserDefaults with the specified key.
56pub fn defaults_get_map(key: &str) -> Option<HashMap<String, String>> {
57	let keys = defaults_get::<Vec<String>>(&format!("{key}.keys"))?;
58	let values = defaults_get::<Vec<String>>(&format!("{key}.values"))?;
59	Some(keys.into_iter().zip(values).collect())
60}
61
62/// Returns the UserDefaults value associated with the specified key, deserialized into a JSON object.
63#[cfg(feature = "json")]
64pub fn defaults_get_json<Value: DeserializeOwned>(key: &str) -> super::error::Result<Value> {
65	let data: String = defaults_get(key).unwrap_or_default();
66	let value = serde_json::from_slice(&data.as_bytes())
67		.map_err(|_| super::error::AidokuError::JsonParseError)?;
68	Ok(value)
69}

71/// Sets the UserDefaults value of the specified key.
72pub fn defaults_set(key: &str, value: DefaultValue) {
73	let value_ptr: Ptr = unsafe {
74		match value {
75			DefaultValue::Bool(ref value) => encode(value),
76			DefaultValue::Int(ref value) => encode(value),
77			DefaultValue::Float(ref value) => encode(value),
78			DefaultValue::String(ref value) => encode(value),
79			DefaultValue::StringArray(ref value) => encode(value),
80			DefaultValue::Null => 0,
81			DefaultValue::HashMap(value) => {
82				// handle hashmap specially
83				// keys and values are stored separately as string arrays
84				let keys = value.keys().cloned().collect::<Vec<_>>();
85				let values = keys
86					.iter()
87					.map(|k| value.get(k).unwrap().clone())
88					.collect::<Vec<_>>();
89				defaults_set(&format!("{key}.keys"), DefaultValue::StringArray(keys));
90				defaults_set(&format!("{key}.values"), DefaultValue::StringArray(values));
91				return;
92			}
93		}
94	};
95	let kind = value.as_byte();
96	unsafe {
97		set(key.as_ptr(), key.len(), kind, value_ptr);
98	};
99	if value_ptr != 0 {
100		unsafe { free_result(value_ptr) };
101	}
102}

104/// Sets the UserDefaults value of the specified key with serialized data.
105pub fn defaults_set_data<T: Serialize>(key: &str, value: T) {
106	let value_ptr: i32 = unsafe { encode(&value) };
107	unsafe {
108		set(key.as_ptr(), key.len(), 0, value_ptr);
109	};
110	unsafe { free_result(value_ptr) };
